function setGlobal(val)
{
	println("Setting global variable: " + val);
	global g_var;
	g_var = val;
}

function display()
{
	println("Before global: " + g_var);
	global g_var;
	println("After global: " + g_var);
}

function main(argv)
{
	setGlobal("bagr");
	display();

	setGlobal(3.14);
	display();
}
