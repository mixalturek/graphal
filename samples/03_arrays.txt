function main(argv)
{
	num = argv.size();
	for(i = 0; i < num; i++)
		echo("argv[" + i + "]: " + argv[i] + "\n");

	arr = array(5);
	arr[0] = "bagr";
	arr[1] = 5.5;
	arr[4] = 3;

	for(i = 0; i < 5; i++)
		echo("arr[" + i + "]: " + arr[i] + "\n");

	st = struct();
	st.bagr = "bagr";
	st.brm = true;

	echo("st.bagr: " + st.bagr + "\n");
	echo("st.brm: " + st.brm + "\n");
	st.bagr = "jiny bagr";
	echo("st.bagr: " + st.bagr + "\n");
}
